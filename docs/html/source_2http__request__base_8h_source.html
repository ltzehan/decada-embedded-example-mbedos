120  HttpResponse* send(Callback<const void*(uint32_t*)> body_cb) {
121 
122  nsapi_error_t ret;
123 
124  if ((ret = connect_socket()) != NSAPI_ERROR_OK) {
125  _error = ret;
126  return NULL;
127  }
128 
129  _request_buffer_ix = 0;
130 
131  set_header("Transfer-Encoding", "chunked");
132 
133  uint32_t request_size = 0;
134  char* request = _request_builder->build(NULL, 0, request_size);
135 
136  // first... send this request headers without the body
137  nsapi_size_or_error_t total_send_count = send_buffer(request, request_size);
138 
139  if (total_send_count < 0) {
140  free(request);
141  _error = total_send_count;
142  return NULL;
143  }
144 
145  // ok... now it's time to start sending chunks...
146  while (1) {
147  uint32_t size;
148  const void *buffer = body_cb(&size);
149 
150  if (size == 0) break;
151 
152  // so... size in HEX, \r\n, data, \r\n again
153  char size_buff[10]; // if sending length of more than 8 digits, you have another problem on a microcontroller...
154  int size_buff_size = sprintf(size_buff, "%X\r\n", static_cast<size_t>(size));
155  if ((total_send_count = send_buffer(size_buff, static_cast<uint32_t>(size_buff_size))) < 0) {
156  free(request);
157  _error = total_send_count;
158  return NULL;
159  }
160 
161  // now send the normal buffer... and then \r\n at the end
162  total_send_count = send_buffer((char*)buffer, size);
163  if (total_send_count < 0) {
164  free(request);
165  _error = total_send_count;
166  return NULL;
167  }
168 
169  // and... \r\n
170  const char* rn = "\r\n";
171  if ((total_send_count = send_buffer((char*)rn, 2)) < 0) {
172  free(request);
173  _error = total_send_count;
174  return NULL;
175  }
176  }
177 
178  // finalize...?
179  const char* fin = "0\r\n\r\n";
180  if ((total_send_count = send_buffer((char*)fin, strlen(fin))) < 0) {
181  free(request);
182  _error = total_send_count;
183  return NULL;
184  }
185 
186  free(request);
187 
188  return create_http_response();
189  }

256  nsapi_size_or_error_t send_buffer(char* buffer, uint32_t buffer_size) {
257  nsapi_size_or_error_t total_send_count = 0;
258  while (total_send_count < buffer_size) {
259 
260  // get a slice of the buffer
261  char *buffer_slice = buffer + total_send_count;
262  uint32_t buffer_slice_size = buffer_size - total_send_count;
263 
264  // if request buffer was set, copy it there
265  if (_request_buffer != NULL && _request_buffer_ix + buffer_slice_size < _request_buffer_size) {
266  memcpy(_request_buffer + _request_buffer_ix, buffer_slice, buffer_slice_size);
267  _request_buffer_ix += buffer_slice_size;
268  }
269 
270  nsapi_size_or_error_t send_result = _socket->send(buffer_slice, buffer_slice_size);
271 
272  if (send_result < 0) {
273  total_send_count = send_result;
274  break;
275  }
276 
277  if (send_result == 0) {
278  break;
279  }
280 
281  total_send_count += send_result;
282  }
283 
284  return total_send_count;
285  }

287  HttpResponse* create_http_response() {
288  // Create a response object
289  _response = new HttpResponse();
290  // And a response parser
291  HttpParser parser(_response, HTTP_RESPONSE, _body_callback);
292 
293  // Set up a receive buffer (on the heap)
294  uint8_t* recv_buffer = (uint8_t*)malloc(HTTP_RECEIVE_BUFFER_SIZE);
295 
296  // Socket::recv is called until we don't have any data anymore
297  nsapi_size_or_error_t recv_ret;
298  while ((recv_ret = _socket->recv(recv_buffer, HTTP_RECEIVE_BUFFER_SIZE)) > 0) {
299 
300  // Pass the chunk into the http_parser
301  uint32_t nparsed = parser.execute((const char*)recv_buffer, recv_ret);
302  if (nparsed != recv_ret) {
303  // printf("Parsing failed... parsed %d bytes, received %d bytes\n", nparsed, recv_ret);
304  _error = -2101;
305  free(recv_buffer);
306  return NULL;
307  }
308 
309  if (_response->is_message_complete()) {
310  break;
311  }
312  }
313  // error?
314  if (recv_ret < 0) {
315  _error = recv_ret;
316  free(recv_buffer);
317  return NULL;
318  }
319 
320  // When done, call parser.finish()
321  parser.finish();
322 
323  // Free the receive buffer
324  free(recv_buffer);
325 
326  if (_we_created_socket) {
327  // Close the socket
328  _socket->close();
329  }
330 
331  return _response;
332  }
